for(obj_name in ls()){
    obj <- get(obj_name)

    for(i in 1:ncol(obj)){
        if(is.numeric(obj[, i])){
            obj[, i] <- round(obj[, i], 3)
        }
    }

    assign(obj_name, obj)
}
