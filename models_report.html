Rounding values

Code 
for(obj_name in c("connection_sh_model", "internet_sh_model")){
    obj <- get(obj_name)

    for(i in 1:ncol(obj)){
        if(is.numeric(obj[, i])){
            obj[, i] <- round(obj[, i], 3)
        }
    }

    assign(obj_name, obj)
}
